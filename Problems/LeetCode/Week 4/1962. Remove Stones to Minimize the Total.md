class Solution {

public:

    int minStoneSum(vector<int>& piles, int k) {
        priority_queue<int> pq;

        for(int i=0; i < piles.size(); i++) {
            pq.push(piles[i]);
        }

        for(int i=0; i < k; i++) {
            int currLargest = pq.top();
            pq.pop();
            pq.push(currLargest - currLargest / 2);
        }
  
        int minTotal = 0;
  
        while(!pq.empty()){
            minTotal += pq.top();
            pq.pop();
        }
        return minTotal;
    }
};
