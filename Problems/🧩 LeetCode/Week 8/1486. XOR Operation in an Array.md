```cpp
class Solution {
public:
    int xorOperation(int n, int start) {
        vector<int> arr(n, 0);

        for(int i=0; i < n; i++) {
            arr[i] = start + 2 * i;
        }

        int result = 0;
        for (int num : arr) {
            result ^= num; // XOR with cumulative result
        }
        return result;
    }
};
```
